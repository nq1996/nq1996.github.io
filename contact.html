# 开发背景

符合自己项目案例所借鉴，于是就打算自己手写一个。以下是原生
js写的案例，希望大家能从中获取灵感。

	<script src="./lib/echarts.min.js"></script>
	<script src="lib/jquery.min.js"></script>
		// 设置节点(图上的矩形,即后面你要布局的元素)宽高，为固定值 假设为 200w,100h
		// 假设节点为3行3列
		const NODE_WIDTH = 200
		const NODE_HEIGHT = 100
		const ROW_NUM = 2
		const COL_NUM = 3

		// 选中echarts实例挂载的对象元素获取宽高
		let boxElWidth = $('#lines_box')[0].offsetWidth
		let boxElHeight = $('#lines_box')[0].offsetHeight
		let linesChart = echarts.init($('#lines_box')[0])

		// 计算布局内(echarts实例容器)x方向间隔y方向间隔
		let stepX = Math.floor((boxElWidth - NODE_WIDTH * COL_NUM) / (COL_NUM - 1))
		let stepY = Math.floor((boxElHeight - NODE_HEIGHT * ROW_NUM) / (ROW_NUM - 1))
		console.log(stepX,stepY,'<===X,Y方向间隔')

		// 选中所有的节点即要布局的元素
		let nodes = $('.node')
		// 此处是为了给节点做布局,只是一种思路,具体根据自己的项目来,
		Array.from(nodes).forEach(el => {
			let target = el.innerHTML
			switch (target) {
				case '11':
					el.style.cssText = `left:${getCoords(1,1)[0]}px;
					top:${getCoords(1,1)[1]}px`
					break;
				case '12':
					el.style.cssText = `left:${getCoords(1,2)[0]}px;
					top:${getCoords(1,2)[1]}px`
					break;
				case '13':
					el.style.cssText = `left:${getCoords(1,3)[0]}px;
					top:${getCoords(1,3)[1]}px`
					break;
				case '21':
					el.style.cssText = `left:0px;
					top:${getCoords(2,1)[1]}px`
					break;
				case '22':
					el.style.cssText = `left:${getCoords(2,2)[0]}px;
					top:${getCoords(2,2)[1]}px`
					break;
				case '23':
					el.style.cssText = `left:${getCoords(2,3)[0]}px;
					top:${getCoords(2,3)[1]}px`
					break;
				default:
					break;
			}
		})
		
		linesChart.setOption({
			xAxis: {
				show: false,
				min: 0,
				// 重点目的是为了让echarts实例x轴长度与容器保持一致
				max: boxElWidth
			},
			yAxis: {
				show: false,
				min: 0,
				// 重点目的是为了让echarts实例y轴长度与容器保持一致
				max: boxElHeight
			},
			/**
			 	* ☆☆☆重点之处,一定要四个方向上都设为0,目的是让节点的
				* 坐标和echarts实例对象的坐标系统相统一,方便定位
			*/
			grid: {
				left: 0,
				top: 0,
				right: 0,
				bottom: 0,
				// 不要包括刻度线宽度
				containLabel: false
			},
			series: [
        {
          type: 'lines',
          coordinateSystem: 'cartesian2d',
					// 线两端的形状控制
					symbol: ['arrow', 'arrow'],
					// 要移动的元素
					effect: {
						show: true,
						symbolSize: [60,80],
						trailLength: 0,
						// loop: false,
						symbol: 'image://' + // 你的图片地址
					},
          data: [
						// 表示每一段线坐标,线坐标有了机器人才能去动起来
						// 反向运动只要把坐标反向设置就行
						{
							coords:	[
								[NODE_WIDTH, NODE_HEIGHT / 2],
								[NODE_WIDTH + stepX, NODE_HEIGHT / 2]
							]
						},
						{
							coords:	[
								[NODE_WIDTH + stepX, NODE_HEIGHT / 2],
								[NODE_WIDTH, NODE_HEIGHT / 2],
							]
						}
					],
          silent: true,
          lineStyle: {
            color: '#27ae60',
            opacity: 1,
            width: 4
          }
        }
      ]
		})
		// 封装坐标函数
		function getCoords(x,y) {
			return [
				(y - 1) * NODE_WIDTH + (y - 1) * stepX,
				(x - 1) * NODE_HEIGHT + (x - 1) * stepY
			]
		}
